int calculer(int a, int b) {
    int resultat;
    
    switch (a) {
        case 0:
            resultat = b;
            break;
        
        case 1:
            resultat = a + b;
            break;
        
        case 2:
            resultat = a * b;
            break;
        
        default:
            resultat = a - b;
            break;
    }
    
    return resultat;
}
