int main() {
    int choix;
    int resultat;
    float prix;
    char ouaret;
    
    // Affectations
    choix = 2;
    resultat = 0;
    prix = 19.99;
    
    // Test switch/case
    switch (choix) {
        case 1:
            resultat = 10 + 5;
            prix = prix * 2;
            break;
        
        case 2:
            resultat = 20 - 3;
            prix = prix / 2;
            ouaret = 'A';
            break;
        
        case 3:
            resultat = 5 * 4;
            break;
        
        default:
            resultat = 0;
            prix = 0.0;
            break;
    }
    
    // Incrémentation et décrémentation
    resultat++;
    --resultat;
    
    return 0;
}
